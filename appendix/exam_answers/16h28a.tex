\paragraph{3. (\hyperref[subsec:能量]{能量}, \hyperref[subsec:动量]{动量})}

\begin{equation*}
    \frac12mv^2=mgH\implies
    H=\frac{v^2}{2g}
\end{equation*}

\begin{gather*}
    mv=(m+M)v'\\
    v'=\frac{m}{m+M}v\\
    \frac12mv^2=mgh+\frac12(m+M)\left(\frac{m}{m+M}v\right)^2\\
    h=\frac{M}{m+M}\frac{v^2}{2g}
\end{gather*}
